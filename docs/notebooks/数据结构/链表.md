## 链表排序

### 归并排序

```js
var sortList = function(head) {
    return mergeSort(head, null)
};

function mergeSort(head, tail) {
    if(head === null) return head

    if(head.next === tail) {
        head.next = null
        return head
    }
    let slow = head, fast = head
    while(fast !== tail) {
        slow = slow.next
        fast = fast.next
        if(fast !== tail) {
            fast = fast.next
        }
    }
    const mid = slow
    return merge(mergeSort(head, mid), mergeSort(mid, tail))
}

function merge(head1, head2) {
    const dummy = new ListNode(0)
    let temp = dummy, temp1 = head1, temp2 = head2
    while(temp1 !== null && temp2 !== null) {
        if(temp1.val < temp2.val) {
            temp.next = temp1
            temp1 = temp1.next
        }else {
            temp.next = temp2
            temp2 = temp2.next
        }
        temp = temp.next
    }
    temp.next = temp1 === null ? temp2 : temp1

    return dummy.next
}
```

## 回文链表

```js
function reverseList(head) {
    let prev = null
    let dummy = head
    while(dummy) {
        let cur = dummy.next
        dummy.next = prev
        prev = dummy
        dummy = cur
    }

    return prev
}
function getMidNode(head) {
    let slow = head, fast = head
    while(fast.next !== null && fast.next.next !== null) {
        slow = slow.next
        fast = fast.next.next
    }

    return slow
}
var isPalindrome = function(head) {
    if(head === null) return true

    let midNode = getMidNode(head)
    let endList = reverseList(midNode.next)
    let result = true
    let head1 = head, head2 = endList
    while(result && head2 !== null) {
        if(head1.val !== head2.val) result = false
        head1 = head1.next
        head2 = head2.next
    }
    midNode.next = reverseList(endList)
    return result
};
```
